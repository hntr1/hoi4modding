country_event = {
	id = eliteformations.1
	title = eliteformations.1.t
	desc = eliteformations.1.d

	picture = GFX_lssah_event_picture

	is_triggered_only = yes

	option = {
		name = eliteformations.1.a

		random_owned_controlled_state = {
			limit = { ROOT = { has_full_control_of = PREV } }
			create_unit = { # Spawn LSSAH regiment somewhere in Germany. Division starts fully equipped with veteran status.
				division = "name = \"1. SS-Division Liebstandarte SS Adolf Hitler\" division_template = \"LSSAH\" start_experience_factor = 1.0 start_equipment_factor = 1"
				owner = GER
			}
		}
	}
}
